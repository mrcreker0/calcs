\begin{theorem}

    $$
        f \in R[a,b] \Rightarrow \in \mathbb{C}[a,b] 
    $$

    \begin{proof}
        $ \forall  x \in [a,b]$ x - фиксированное число
        $x + n \in[a,b] $, где n - приращение
        $ \Delta F = F(x + n) - F(x) = \int\limits_a^{x+n} f(t) \, dt - \int\limits_a^x f(t) \, dt = \int\limits_x^{x+n} f(t) \, dt $ \\
        $ f \in R[a.b] f$ - ограниченная функция, значит $ \exists M \geqslant 0: |f(x)| \leqslant M, \forall x \in [a,b] $ \\
        $ 0 \leqslant |\Delta F(x)| = |\int\limits_x^{x+n} f(t) \, dt| \leqslant k * |n| \space (|n| \to 0) \Rightarrow \lim\limits_{n \to 0} \Delta F(x) = 0 \overset{\text{опр}}{\Leftrightarrow}$ % оверсет опр
        $ F \in \mathbb{C}[a,b] $
    \end{proof}
\end{theorem}

\section{Формула Ньютона-Лейбница(Ф.Н-Л)}
% \text{будет дорабатываться}
\begin{theorem}
    $ f \in [a,b] \Rightarrow | \int\limits_{a}^{b} f(t) \, dt = \varphi(b) - \varphi(a) \, $ где $\varphi$ - первообразная для $f$ на $[a,b] $ \\
    \begin{proof}
        $ f \in \mathbb{C}[a,b] \Rightarrow F(x) = \int\limits_{a}^{x} f(t) \, dt $ - первообразная для $f$ \\
        $ \varphi(x)$ - тоже первообразная для $f \Rightarrow \varphi(x) = F(x)+C, C - \text{константа} $ \\
        пусть $ x = a: \varphi(a) = F(a) + C = C $ \\
        $ F(x) = \int\limits_{a}^{x} f(t) \, dt = \varphi(x) - \varphi(a) $ \\
        пусть $ x = b: \int\limits_{a}^{b} f(t) \, dt = \varphi(b) - \varphi(a) $
    \end{proof} 
\end{theorem}

\section{Замена переменной в определенном интеграле}
% \text{будет доррабатываться}
\begin{lemma}
    Пусть $f \in C[a,b]$, $\varphi \in C^1[\alpha,\beta]$, 
    $\varphi(\alpha)=a$, $\varphi(\beta)=b$. Тогда
    \[
    \int\limits_a^b f(x)\,dx = \int\limits_{\alpha}^{\beta} f(\varphi(t)) \varphi'(t)\,dt.
    \]

    \begin{proof}
        Так как $f \in C[a,b]$, существует первообразная $F$ такая, что $F'(x)=f(x)$. Тогда по формуле Ньютона--Лейбница:
        \[
        \int\limits_a^b f(x)\,dx = F(b)-F(a).
        \]

        Рассмотрим функцию $F(\varphi(t))$. По правилу цепочки:
        \[
        \frac{d}{dt} F(\varphi(t)) = F'(\varphi(t)) \cdot \varphi'(t) = f(\varphi(t)) \varphi'(t).
        \]

        Следовательно, $F(\varphi(t))$ является первообразной функции 
        $f(\varphi(t))\varphi'(t)$, и снова по формуле Ньютона--Лейбница:
        \[
        \int\limits_{\alpha}^{\beta} f(\varphi(t)) \varphi'(t)\,dt
        = F(\varphi(\beta)) - F(\varphi(\alpha)).
        \]

        С учётом условий $\varphi(\alpha)=a$, $\varphi(\beta)=b$, получаем:
        \[
        F(\varphi(\beta)) - F(\varphi(\alpha)) = F(b)-F(a) \Rightarrow
        \int\limits_a^b f(x)\,dx = \int\limits_{\alpha}^{\beta} f(\varphi(t)) \varphi'(t)\,dt.
        \]
    \end{proof}

\end{lemma}
